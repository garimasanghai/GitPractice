mkdir <- to make a folder
ls <- to see whatever is in the folder
touch <- to create a file
.txt / .py <- extension u need to add at th end of the file
git init <- to initialize a folder as a git repository
cd .. <- to get outside
cd ./foldername <- to get inside a folder (./ <--- used for absolute path)
cd foldername <- (for relative path ./ isn't required)
.gitignore <- to specify & list down which files git ignores
rm ./foldername <- remove the folder
rm ./fileName.ext <-remove the file
git add filename.ext <- add the specific files in the staging area
git add . <- add all the files in the staging area (. means the current folder)
git status <-status of the git files which git is keeping tabs
git commit -m "added" <- commit the code into the local repository with a heading
git push ,- pushing the code into the repo
git branch <- check how many branches r there, their names, & the current branch u r in
git branch -M main <- where M makes any branch the default branch, in this case main is the name of the branch which we r making default
git branch branchName <- to create a new branch from the old branch
git checkout branchName <- to go to the specified branch
git stash <- keeping the changes for later use
git stash pop <- to get back the stashed changes. [takes out the changes out of the stash rack]
